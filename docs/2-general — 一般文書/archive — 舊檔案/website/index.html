// 計算平靜指數
import { calculateCalmIndex } from '@moltos/core';

const result = await calculateCalmIndex({
  userId: 'user_abc',
  window: 14,      // 14 天滾動視窗
  dimensions: [
    'messageVolume',
    'replyLatency',
    'nightActivity',
    'unreadPileup',
    'voiceEmotion',
  ],
});

console.log(result.score);  // 82
console.log(result.level);  // "Calm"
